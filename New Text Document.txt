# OOPDB
OOPDB - OOP Data Base wrapper based on sqlite3 python module

## Example
```
# Opening db
db = OOPDB("some_db.db")

# Table creation
# PrimaryKey class used for primary key column configuration
id = PrimaryKey(name="Id", is_auto_increment=True)
some_data = ColumnConfig(name="some_data", is_null=False, type=DataTypes.TEXT)
db.create_table("test_table", [id, some_data])

# inserting data
db.insert_into("test_table", ["Id", "some_data"], [1, "SomeDataTextContent"]).execute()

# or queue several commands
db.insert_into("test_table", ["Id", "some_data"], [2, "SomeDataTextContent2"])
db.insert_into("test_table", ["Id", "some_data"], [3, "SomeDataTextContent3"])
# and then push them to the data base
db.execute()

# selecting data
data = db.select("test_table", ["some_data"]).fetch()

# close connection
db.close()
```
